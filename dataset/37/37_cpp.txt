#include<iostream>
#include<list>
int main()
{
    std::list<int> lt;

    for(int i=0;i<10;i++)
    {
        lt.push_back(i);
    }

    std::list<int>::iterator p = lt.begin();
    std::cout<<"List : ";
    while(p!=lt.end())
    {
        std::cout<<*p<<" ";
        p++;
    }
    std::cout<<std::endl;

    p = lt.end();
    std::cout<<"List : ";
    while(p!=lt.begin())
    {
        p--;                  //decrementing so that p should first point to the last element 
        std::cout<<*p<<" ";
    }
    std::cout<<std::endl;

    return 0;
}
