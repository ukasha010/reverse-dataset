#include<iostream>
#include<fstream>       
using namespace std;
int main()
{
    ifstream in("Data.txt");       //open file for output(writing)

    if(!in)
    {
        cout<<"Cannot open file!!"<<endl;
        return 1;
    }
    char str[20];

    while(in)
    {
        in>>str;
        cout<<str<<endl;
    }
    
    in.close();
    return 0;
}
