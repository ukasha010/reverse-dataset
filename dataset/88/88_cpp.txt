#include<iostream>
#include<cstring>
using namespace std;

class MyException
{
    public:
        char str[80];
        int num;
        MyException()
        {
            *str=0;
            num=0;
        }
        MyException(char *s,int n)
        {
            strcpy(str,s);
            num=n;
        }
};

int main()
{

    try 
    {
        int n;
        cout<<"Enter a positive number : ";
        cin>>n;

        if(n<0)
        {
            throw MyException("Not positive",n);
        }

        cout<<"Positive number : "<<n<<endl;
    }
    catch(MyException e)
    {
        cout<<e.str<<endl;
        cout<<e.num<<endl;
    }
    return 0;
}
