#include <iostream>
using namespace std;
void divide(double a, double b);

int main()
{
    double a, b;
    cout << "Enter a : ";
    cin >> a;
    cout << "Enter b : ";
    cin >> b;
    divide(a, b);

    return 0;
}

void divide(double a, double b)
{
    try         //starting of try block
    {
        if (!b)
            throw b;        //throwing an exception if denominator is zero
        cout<<"Result : "<<a/b<<endl;
    }
    catch(double d)
    {
        cout<<"Cannot divide by zero!!"<<endl;
    }
}
