#include <iostream>
#include <vector>
#include <cctype> //declares a set of functions which can classify and transform individual characters
int main()
{
    std::vector<char> ch(5);   //created a vector of length 5 and type char

    std::cout<<"Size of the vector : "<<ch.size()<<std::endl;

    for(int i=0;i<ch.size();i++)
    {
        ch[i] = i + 'a';
    }

    std::cout<<"Vector : ";
    for(int i=0;i<ch.size();i++)
    {
        std::cout<<ch[i]<<" ";
    }
    std::cout<<std::endl;

    for(int i=0;i<10;i++)
    {
        ch.push_back(i+5+'a');
    }

    std::cout<<"Vector : ";
    for(int i=0;i<ch.size();i++)
    {
        std::cout<<ch[i]<<" ";
    }
    std::cout<<std::endl;


    for(int i=0;i<ch.size();i++)
    {
        ch[i] = toupper(ch[i]);
    }

    std::cout<<"Vector : ";
    for(int i=0;i<ch.size();i++)
    {
        std::cout<<ch[i]<<" ";
    }
    std::cout<<std::endl;

    return 0;

}
