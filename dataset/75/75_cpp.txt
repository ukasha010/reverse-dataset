#include<iostream>
using namespace std;
template<class X> void swapArgs(X &a,X &b)      //also known as template function
{
    X temp;
    temp = a;
    a = b;
    b = temp;
}
int main()
{
    int i=5,j=20;
    double x=11,y=23.34;
    char m='c',n='d';
    cout<<"Original : ";
    cout<<"\ni : "<<i<<"\nj : "<<j<<endl;
    cout<<"\nx : "<<x<<"\ny : "<<y<<endl;
    cout<<"\nm : "<<m<<"\nn : "<<n<<endl;

    swapArgs(i,j);
    swapArgs(x,y);
    swapArgs(m,n);
    cout<<"\n\nSwapped Values : ";
    cout<<"\ni : "<<i<<"\nj : "<<j<<endl;
    cout<<"\nx : "<<x<<"\ny : "<<y<<endl;
    cout<<"\nm : "<<m<<"\nn : "<<n<<endl;

    return 0;
}
