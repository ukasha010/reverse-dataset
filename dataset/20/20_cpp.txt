#include<iostream>
using namespace std;
int main()
{
    int a[10];

    cout<<"Input 10 elements : "<<endl;
    for(int i=0;i<10;i++)
    {
        cout<<"Enter element "<<i+1<<" : ";
        cin>>a[i];
    }

    int large=-10000000;
    for(int i=0;i<10;i++)
    {
        if(a[i]>large)
            large=a[i];
    }

    cout<<"Largest element in the array : "<<large<<endl;
    cout<<"Using library function : "<<max(a[0],*(a+10))<<endl;
    return 0;
}
