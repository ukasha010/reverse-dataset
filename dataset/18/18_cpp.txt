#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    string str = "I love music";

    cout<<"Length of the string is : "<<str.size()<<endl;

    char ch[] = "I play piano";
    cout<<"Length of the string is : "<<strlen(ch)<<endl;

    int len=0;
    for(int i=0;ch[i]!='\0';i++)
    {
        len++;
    }
    cout<<"Length of the string is : "<<len<<endl;
    return 0;
}
